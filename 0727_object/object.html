Q. 함수 호출시 new 를 붙인 것과 안 붙인 것의 차이를 설명하세요.


        var obj = function () {
          console.log(this);
          return 10;
        };
        
        - 해당 함수 실행
        - 리턴값인 10을 반환
        - obj()의 타입은 number
        obj();   // Window
        console.log(obj();)   // 10
        
        - 객체를 새로 생성 
        - this에 생성한 객체를 할당하고 해당 함수 실행
        - 생성된 객체(=인스턴스)를 반환
        - 리턴값인 10은 무시됨 
        - new obj()의 타입은 obj
        new obj();  // obj{}
        console.log(new obj();)  // obj{}

Q. 객체 내에서의 this 를 설명하세요.


        var obj = {
          thisIs : this,           
          init : function(){
            console.log(this);    // Object(=obj)
          }
        }

        console.log(obj.thisIs);  // Window
        obj.init();

        init 메서드 내의 this는 obj객체가 할당되고 
        메서드가 아닌 일반 객체에서의 this는 전역 객체(최상위 객체)인 window를 나타낸다.

Q. 데이터 타입들을 모두 넣은후, 값들을 출력해보세요.


        var obj = {
           func : function () {
              console.log('func');
           },
           num : 3,
           str : 'kyhim',
           arr : ['황금올리브','굽네오리지널','허니콤보'],
           und : undefined,
           nothing : null,
           o : {
            name : '치즈스노윙',
            price : 16000}
        };
        
        console.log(obj.func);
        // function () {
              console.log('func');
           }
        
        console.log(obj.num);
        // 3
        
        console.log(obj.str);
        // kyhim
        
        console.log(obj.arr);
        // (3)["황금올리브", "굽네오리지널", "허니콤보"]
        
        console.log(obj.und);
        // undefined
        
        console.log(obj.nothing);
        // null
        
        console.log(obj.o);
        // Object {name: "치즈스노윙", price: 16000}
