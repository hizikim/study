Q. 함수 호출시 new 를 붙인 것과 안 붙인 것의 차이를 설명하세요.


        var obj = function () {
          console.log(this);
          return 10;
        };
        
        - 해당 함수 실행
        - 리턴값인 10을 반환
        - obj()의 타입은 number
        obj();   // Window
        console.log(obj();)   // 10
        
        - 객체를 새로 생성 
        - this에 생성한 객체를 할당하고 해당 함수 실행
        - 생성된 객체(=인스턴스)를 반환
        - 리턴값인 10은 무시됨 
        - new obj()의 타입은 obj
        new obj();  // obj{}
        console.log(new obj();)  // obj{}

Q. 객체 내에서의 this 를 설명하세요.


        var obj = {
          thisIs : this,           
          init : function(){
            console.log(this);    // Object(=obj)
          }
        }

        console.log(obj.thisIs);  // Window
        obj.init();

        init 메서드 내의 this는 obj객체가 할당되고 
        메서드가 아닌 일반 객체에서의 this는 전역 객체(최상위 객체)인 window를 나타낸다.

Q. 객체내의 모든 속성들의 값을 배열에 저장한후, abc 순으로 sorting 하여, 'abcdefg' 문자열로 출력해보세요.


        var obj = {
          e : 'e',
          d : 'd',
          b : 'b',
          g : 'g',
          c : 'c',
          f : 'f',
          a : 'a'
        };
        
        var objArr = $.map(obj, function(val, key){return val});  // obj 객체의 키값(value)를 요소로 갖는 배열
        console.log(objArr);   // (7) ["e", "d", "b", "g", "c", "f", "a"]

        // 배열 요소가 숫자가 아닌 문자이기 때문에 부등호로 비교
        objArr.sort(function(a, b){
           // 앞에서부터 요소 비교. 참일때 자리 바꿈 (큰 수가 뒤로 감)
           if (a > b){
              // console.log(a > b, a, b);
              return true;
            } else {
              // console.log(a > b, a, b);
              return false;
            }
            // return a > b ? true : false;   // 삼항연산자로 쓸 수도 있음
          });
          console.log(objArr);  // (7) ["a", "b", "c", "d", "e", "f", "g"]
          var arrToStr = objArr.join(''); // 배열의 요소들을 문자열로 합침
          console.log(arrToStr);  // abcdefg
