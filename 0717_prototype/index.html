Array.prototype

Array.join()
배열의 모든 요소를 연결해 하나의 문자열로 만듦.
		
		<script>
		var a = ['바람', '비', '불'];
		var myVar1 = a.join();      // myVar1에 '바람,비,불'을 대입
		var myVar2 = a.join(', ');  // myVar2에 '바람, 비, 불'을 대입
		var myVar3 = a.join(' + '); // myVar3에 '바람 + 비 + 불'을 대입
		var myVar4 = a.join('');    // myVar4에 '바람비불'을 대입
		console.log(myVar1);
		</script>
Array.reverse()
배열 내 요소들의 순서를 반전시킴.
원본 배열이 변형됨. 배열의 참조를 반환할 수 있음
		
		var myArray = ['one', 'two', 'three'];
		var myArray2 = myArray.reverse(); // 배열의 참조를 새 변수에 반환 
		
		console.log(myArray) // ['three', 'two', 'one']
Array.sort()
배열의 내 요소들의 순서를 오름차순으로 정렬
원본 배열이 변형됨. 배열의 참조를 반환할 수 있음
숫자 정렬 : compare(a,b){if(areturn a-b}
'-' 연산자로 인자를 빼서 반환값이 0보다 큰지 작은지로 비교하기 때문. 따라서 뺄수 없으면 비교가 안된당~~~
문자열 정렬 : compare(a,b){if(a>b)return a-b} : 숫자만 가능
		
		var fruit = ['cherries', 'apples', 'bananas'];
		var fruit2 = fruit.sort(); // ['apples', 'bananas', 'cherries']
		
		var scores = [1, 10, 21, 2]; 
		scores.sort(); // [1, 10, 2, 21]
		// 10이 2,
		// Unicode 코드 포인트 순서에서 '10'이 '2'앞에 오기 때문에.
		
		var things = ['word', 'Word', '1 Word', '2 Words'];
		things.sort(); // ['1 Word', '2 Words', 'Word', 'word']
		// 유니 코드에서 숫자는 대문자 앞에옵니다.
		// 소문자 앞에옵니다.
